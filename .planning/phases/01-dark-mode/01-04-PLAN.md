**ScrollProgress.tsx - Full Rewrite:**

The current component uses inline `backgroundColor: color` style. Convert to use Tailwind classes with dark: variants.

```tsx
"use client";

import React from "react";
import { motion, useScroll, useSpring } from "framer-motion";

interface ScrollProgressProps {
  height?: number;
  zIndex?: number;
}

const ScrollProgress: React.FC<ScrollProgressProps> = ({
  height = 2,
  zIndex = 100,
}) => {
  const { scrollYProgress } = useScroll();

  // Add spring physics for a more premium feel
  const scaleX = useSpring(scrollYProgress, {
    stiffness: 100,
    damping: 30,
    restDelta: 0.001,
  });

  return (
    <motion.div
      className="fixed top-0 left-0 right-0 origin-left bg-neutral-900/80 dark:bg-neutral-100/80"
      style={{
        scaleX,
        height,
        zIndex,
        transformOrigin: "left",
      }}
    />
  );
};

export default ScrollProgress;
```

Key changes:
- Remove `color` prop (no longer needed)
- Use Tailwind classes: `bg-neutral-900/80 dark:bg-neutral-100/80`
- Dark gray bar on light background, light gray bar on dark background
- Both have 80% opacity for subtle effect
