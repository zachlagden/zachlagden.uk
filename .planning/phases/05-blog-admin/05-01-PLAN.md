```typescript
import { z } from 'zod'

export const postSchema = z.object({
  title: z.string().min(1, 'Title is required').max(200, 'Title too long'),
  slug: z.string()
    .min(1, 'Slug is required')
    .max(100)
    .regex(/^[a-z0-9-]+$/, 'Slug must be lowercase letters, numbers, and hyphens only'),
  excerpt: z.string()
    .min(10, 'Excerpt must be at least 10 characters')
    .max(500, 'Excerpt too long'),
  content: z.string().min(1, 'Content is required'),
  categories: z.array(z.string()).min(1, 'Select at least one category'),
  tags: z.array(z.string()),
  featuredImage: z.string().url('Must be a valid URL'),
  published: z.boolean()
})

export type PostFormData = z.infer<typeof postSchema>

// For form state management
export type PostFormState = {
  errors?: z.inferFlattenedErrors<typeof postSchema>['fieldErrors']
  message?: string
  success?: boolean
}
```

Include the slug auto-generation helper:
```typescript
export function generateSlug(title: string): string {
  return title
    .toLowerCase()
    .trim()
    .replace(/[^\w\s-]/g, '')
    .replace(/[\s_-]+/g, '-')
    .replace(/^-+|-+$/g, '')
}
```

Create `src/lib/actions/posts.ts` with 'use server' directive:

**createPost action:**
- Call `requireAdmin()` first (from dal.ts)
- Parse FormData and validate with postSchema.safeParse()
- Check slug uniqueness against database
- Insert document with: author='Zach Lagden', previous_slugs=[], readingTime calculated from content, timestamps
- Call `revalidatePath('/blog')` and `revalidatePath('/')` after insert
- Return `{ success: true, slug }` or `{ errors: {...} }`

**updatePost action:**
- Accept postId (string) and FormData
- Call `requireAdmin()` first
- Validate with postSchema
- If slug changed: append old slug to previous_slugs array
- Update document with new data, update `updatedAt` timestamp
- If published changed to true and publishedAt is null, set publishedAt to now
- Call `revalidatePath('/blog')`, `revalidatePath('/blog/${newSlug}')`, `revalidatePath('/blog/${oldSlug}')` if slug changed
- Return `{ success: true, slug }` or `{ errors: {...} }`

**deletePost action:**
- Accept postId (string)
- Call `requireAdmin()` first
- Delete document from collection
- Call `revalidatePath('/blog')` and `revalidatePath('/')`
- Return `{ success: true }` or `{ error: string }`

**togglePublish action:**
- Accept postId (string) and publish (boolean)
- Call `requireAdmin()` first
- Update published field; if setting to true and publishedAt is null, set publishedAt
- Call `revalidatePath('/blog')` and individual post path
- Return `{ success: true }` or `{ error: string }`

Use `calculateReadingTime` from `@/lib/blog/posts` for reading time.
Use `ObjectId` from 'mongodb' for document IDs.

Important patterns:
- Always check `validated.success` before accessing `validated.data`
- Use `redirect()` from 'next/navigation' only after all revalidation (redirect throws)
- Don't redirect from Server Actions - return result and let client redirect
